package atividades_classe_2003;

public class ListaEncadeada {

	public static void main(String[] args) {
		int meu_array[] = new int[10];
		
		for (int i=0;i<meu_array.length;i++) {
			meu_array[i] = i+1;
		};
		for (int i=0;i<meu_array.length;i++) {
			System.out.println(meu_array[i]);
		};		
	}

}
